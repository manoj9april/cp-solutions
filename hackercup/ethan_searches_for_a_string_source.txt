#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string s;
	for(int x=1;x<=t;x++)
	{
		cin>>s;
		int i;
		for(i=1;i<s.size();i++)
		{
			if(s[0]==s[i])
			{
				break;
			}
		}
		if(i==s.size() || i==s.size()-1)
		{
			cout<<"Case #"<<x<<": Impossible"<<endl;
		}
		else if(i==1)
		{
			int flag=0;
			for(int k=1,l=2;l<s.size();k++,l++)
			{
				if(s[k]!=s[l])
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			{
				cout<<"Case #"<<x<<": "<<"Impossible"<<endl;
			}
			else
				cout<<"Case #"<<x<<": "<<s[0]<<s<<endl;
		}
		else
		{
			int flag=0;
			for(int k=1,l=i+1;l<s.size();k++,l++)
			{
				if(s[k]!=s[l])
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			{
				cout<<"Case #"<<x<<": "<<"Impossible"<<endl;
			}
			else
			{
				cout<<"Case #"<<x<<": ";
				for(int k=0;k<i;k++)
				{
					cout<<s[k];
				}
				cout<<s<<endl;
			}
		}
	}
return 0;
}
